Randomise letter size in String

PFont f;

String message = "Randomise size.";

int rand;

void setup() {

  size(600, 400);

  f = createFont("Arial",20,true);

}

void draw() {

  background(255);

  fill(0);

  textFont(f);        

  // Displaying a block of text all at once using text().    

  // text(message,10,height/2);

  // The first character is at pixel 10.

  int x = 10;

  for (int i = 0; i < message.length(); i++) {

   rand = int(random(12,60));

  textSize(rand);

  // Each character is displayed one at a time with the charAt() function.

  text(message.charAt(i),x,height/2);

  // All characters are spaced 10 pixels apart.

  x += 2+rand;

  }

  noLoop();

}

  size(400, 400);

  f = createFont("Arial",20,true);

}

void draw() {

  background(255);

  fill(0);

  textFont(f);                  // Set the font

  translate(width/2,height/2);  // Translate to the center

  rotate(spin);                // Rotate by spin

  textAlign(CENTER);    

  text(message,0,0);            

  spin += float(mouseX-200)/200;        // Increase rotation by mouseX

}

wobblyText (with array)

PFont f;

String message = "click and hold mouse";

// An array of Letter objects

Letter[] letters;

void setup() {

  size(470, 190);

  // Load the font

  f = createFont("Arial",48,true);

  textAlign(CENTER);

  textFont(f);

  // Create the array the same size as the String

  letters = new Letter[message.length()];

  // Initialize Letters at the correct x location

  int x = 16;

  for (int i = 0; i < message.length(); i++) {

    letters[i] = new Letter(x,100,message.charAt(i));

    x += textWidth(message.charAt(i));

  }

}

void draw() {

  background(255);

  for (int i = 0; i < letters.length; i++) {

    // Display all letters

    letters[i].display();

   

    // If the mouse is pressed the letters shake

    // If not, they return to their original location

    if (mousePressed) {

      letters[i].shake();

    } else {

      letters[i].home();

    }

  }

}

// A class to describe a single Letter

class Letter {

  char letter;

  // The object knows its original "home" location

  float homex,homey;

  // As well as its current location

  float x,y;

  Letter (float x_, float y_, char letter_) {

    homex = x = x_;

    homey = y = y_;

    letter = letter_;

  }

  // Display the letter

  void display() {

    fill(0);

    textAlign(LEFT);

    text(letter,x,y);

  }

  // Move the letter randomly

  void shake() {

    x += random(-2,2);

    y += random(-2,2);

  }

 // Return the letter home

  void home() {

    x = homex;

    y = homey;

  }

}

  ellipse ( mouseX, mouseY, 2,2); // draws small circles
